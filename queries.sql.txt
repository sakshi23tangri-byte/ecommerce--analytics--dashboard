--Total Revenue 
SELECT SUM(CAST(payment_value AS NUMERIC)) AS total_revenue
FROM payments;

--Total Orders
SELECT COUNT(*) AS total_orders
FROM orders;

--Total Customers
SELECT COUNT(DISTINCT customer_id) AS total_customers
FROM customers;

-- orderr byy status
SELECT order_status,
       COUNT(*) AS total_orders
FROM orders
GROUP BY order_status
ORDER BY total_orders DESC;

-- Monthly Revenue Trend
SELECT 
    DATE_TRUNC('month', CAST(order_purchase_timestamp AS TIMESTAMP)) AS month,
    SUM(CAST(p.payment_value AS NUMERIC)) AS revenue
FROM orders o
JOIN payments p 
    ON o.order_id = p.order_id
GROUP BY month
ORDER BY month;

--Top 10 customers by spending
SELECT 
    o.customer_id,
    SUM(CAST(p.payment_value AS NUMERIC)) AS total_spent
FROM orders o
JOIN payments p 
    ON o.order_id = p.order_id
GROUP BY o.customer_id
ORDER BY total_spent DESC
LIMIT 10;

--Best selling products
SELECT 
    product_id,
    SUM(CAST(price AS NUMERIC)) AS total_sales
FROM order_items
GROUP BY product_id
ORDER BY total_sales DESC
LIMIT 10;

--Customer Segmentation
SELECT 
    o.customer_id,
    SUM(CAST(p.payment_value AS NUMERIC)) AS total_spent,
    CASE 
        WHEN SUM(CAST(p.payment_value AS NUMERIC)) > 10000 THEN 'High Value'
        WHEN SUM(CAST(p.payment_value AS NUMERIC)) BETWEEN 5000 AND 10000 THEN 'Medium Value'
        ELSE 'Low Value'
    END AS customer_segment
FROM orders o
JOIN payments p 
    ON o.order_id = p.order_id
GROUP BY o.customer_id
ORDER BY total_spent DESC
Limit 10;

__ Rank top customers(using window function) 
SELECT 
    customer_id,
    total_spent,
    RANK() OVER (ORDER BY total_spent DESC) AS customer_rank
FROM (
    SELECT 
        o.customer_id,
        SUM(CAST(p.payment_value AS NUMERIC)) AS total_spent
    FROM orders o
    JOIN payments p 
        ON o.order_id = p.order_id
    GROUP BY o.customer_id
) t;

-- Rank top customers(using window function) 
SELECT 
    customer_id,
    total_spent,
    RANK() OVER (ORDER BY total_spent DESC) AS customer_rank
FROM (
    SELECT 
        o.customer_id,
        SUM(CAST(p.payment_value AS NUMERIC)) AS total_spent
    FROM orders o
    JOIN payments p 
        ON o.order_id = p.order_id
    GROUP BY o.customer_id
) t;

--Revenue Growth
WITH monthly_revenue AS (
    SELECT 
        DATE_TRUNC('month', CAST(order_purchase_timestamp AS TIMESTAMP)) AS month,
        SUM(CAST(p.payment_value AS NUMERIC)) AS revenue
    FROM orders o
    JOIN payments p 
        ON o.order_id = p.order_id
    GROUP BY month
)
SELECT *,
       LAG(revenue) OVER (ORDER BY month) AS previous_month_revenue
FROM monthly_revenue;
